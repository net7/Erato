def self.schema_for(type, context)
  properties = {}
  Talia3.repository.query query(type, context) do |s|
    begin
      key = Talia3::URI.new(s.uri).to_key
      type, object_type = format_range s
      label, language   = format_label s
    
      properties[key] = {:type => type, :object_type => object_type} unless properties[key]
      (properties[key][:label] ||= {})[language] = label if label
    rescue
    end
  end
  properties.blank? ? {} : {:properties => properties, :order => nil}
end
