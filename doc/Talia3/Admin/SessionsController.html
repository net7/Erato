def create
  user = Talia3::Auth.user params[:session]
  if user.nil?
    # TODO: i18n
    flash.now[:error] = "Invalid email/password combination."
    render :new
  else
    log_in user
    # TODO: i18n
    flash.now[:notice] = "Authentication successful."
    redirect_back_or :talia3_admin
  end
end
