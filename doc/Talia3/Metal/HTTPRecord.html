def self.do_rest(env)
  @env = env
  @path_params = env['action_dispatch.request.path_parameters']
  method = @env['REQUEST_METHOD']
  case method
    when 'GET'    then show
    when 'POST'   then create_or_update
    when 'DELETE' then destroy
  end
end
